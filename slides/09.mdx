# 2 built-in ways to style components:

1. Set the class with `className`, and use regular CSS files
```jsx  
import 'styles.css'

function NormalCSS() {
    return (
        // Use className instead of class
        <p className="big-text">
            BIG!
        </p>
    )
}
```

function InlineStyle() {

        // Double {} because you're defining a JS object inside JSX

            fontSize: 20, // Camel case attributes
            zIndex: 2, 
            color: "blue", // values like blue must be in ""

        </p>
